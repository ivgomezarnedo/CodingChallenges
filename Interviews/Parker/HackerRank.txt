RESULT: 
/*
Enter your query below.
Please append a semicolon ";" at the end of the query
*/
SELECT p.pname,
       ROUND( SUM(CASE WHEN g.result='won' THEN 1 ELSE 0 END) / 
       NULLIF( SUM(CASE WHEN g.result='lost' THEN 1 ELSE 0 END), 0), 2) AS ratio
FROM players as p 
LEFT JOIN games as g on p.id=g.id
GROUP BY p.id, p.pname
ORDER BY ratio DESC, pname DESC;

result:

SELECT month, 
    ROUND(
        (monthly_cases / NULLIF(cumulative_cases_prior_month, 0)) * 100,
    1) AS percentage_increase
from (
SELECT MONTH(record_date) as month,
 SUM(cases_count) as monthly_cases,
SUM(SUM(cases_count)) OVER (ORDER BY YEAR(record_date), MONTH(record_date) ROWS BETWEEN UNBOUNDED PRECEDING AND 1 PRECEDING) AS cumulative_cases_prior_month
from daily_cases
group by MONTH(record_date), YEAR(record_date) ) as x
where cumulative_cases_prior_month is not null
order by month asc

import heapq # priority queue

def getMaximumAmount(quantity, m):
    # Write your code here
    # One iteration per customer.
    # Highest available quantity item is sold on each iteration.
    heap_max = [-q for q in quantity if q >0] # Heapq only provides a min heap implementation
    heapq.heapify(heap_max) # Transforms list to heap
    total_revenue = 0
    
    for x in range(m):
        if heap_max:
            current_item = -heapq.heappop(heap_max) # Highest quantity item
            total_revenue += current_item
            if current_item >1: # If there are items left
                # Decreasing quantity (one item is sold) and pushing back to queue
                heapq.heappush(heap_max, -(current_item -1))
    return total_revenue
